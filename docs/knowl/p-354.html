\begin{equation*} \chi _p \colon \QQ_p^\times \to \QQ_p \end{equation*}

\begin{equation*} \chi _p\colon H_f^1(G_{\QQ_p}, \QQ_p(1)) \to \QQ_p \end{equation*}
