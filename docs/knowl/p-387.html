\begin{equation*} H^1_f(G_{K_p}, \QQ_p(1)) \cong \ints_p^\times \otimes \QQ_p \cong \QQ_p \end{equation*}

\begin{equation*} H^1_f(G_{T}, \QQ_p(1)) \cong \ints_K^\times \otimes \QQ_p \cong 0 \end{equation*}

then

\begin{equation*} \dim H^1_f( G_T, \ldots) \lt \dim H_f^1(G_{K_p}, \ldots)\text{.} \end{equation*}
